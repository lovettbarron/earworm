```go
func executeCommand(t *testing.T, args ...string) (string, error) {
    t.Helper()
    viper.Reset()
    t.Cleanup(func() { viper.Reset() })
    // Resets all package-level flag variables
    buf := new(bytes.Buffer)
    rootCmd.SetOut(buf)
    rootCmd.SetErr(buf)
    rootCmd.SetArgs(args)
    err := rootCmd.Execute()
    return buf.String(), err
}
```
