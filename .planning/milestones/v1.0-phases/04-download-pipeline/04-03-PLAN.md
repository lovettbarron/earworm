    ```go
    package download

    // PipelineConfig holds all configuration for the download pipeline.
    type PipelineConfig struct {
        StagingDir        string
        LibraryDir        string
        RateLimitSeconds  int
        MaxRetries        int
        BackoffMultiplier float64
        Quiet             bool
    }

    // Summary holds the results of a pipeline run.
    type Summary struct {
        Total       int
        Succeeded   int
        Failed      int
        Interrupted bool
        AuthFailed  bool
        Elapsed     time.Duration
        Errors      []BookError  // per-book error details
    }

    // BookError records a download failure for the completion summary.
    type BookError struct {
        ASIN    string
        Title   string
        Author  string
        Message string
    }

    // Pipeline orchestrates batch audiobook downloads.
    type Pipeline struct {
        client   audible.AudibleClient
        db       *sql.DB
        config   PipelineConfig
        progress *ProgressTracker
    }

    func NewPipeline(client audible.AudibleClient, database *sql.DB, cfg PipelineConfig, w io.Writer) *Pipeline
    ```

    `Run(ctx context.Context) (*Summary, error)` method:

    1. Call `CleanOrphans(cfg.StagingDir, downloadedASINsFromDB)` -- query DB for downloaded ASINs, pass to CleanOrphans. Log warning on error, don't abort (per D-07).

    2. Call `db.ListDownloadable(p.db)` to get books to download.

    3. If any books have status "downloaded" or "organized" in the full list context, report resume state via `p.progress.FormatResume()` (per D-06). Calculate previously completed by querying count of downloaded books with download_completed_at set.

    4. Create `NewRateLimiter(cfg.RateLimitSeconds)` and `NewBackoffCalculator(cfg.RateLimitSeconds, cfg.BackoffMultiplier)`.

    5. Loop over books:
       a. Check `ctx.Done()` between books -- if cancelled, set `summary.Interrupted = true`, break.
       b. Rate limit wait (skip on first book). If Wait returns ctx.Err(), break.
       c. Call `downloadWithRetry(ctx, book, backoff)`.
       d. If err is `*audible.AuthError`: set `summary.AuthFailed = true`, return with error message "authentication failed: run `earworm auth` to re-authenticate" (per D-10).
       e. If err is non-nil (exhausted retries): increment `summary.Failed`, append to `summary.Errors`.
       f. If err is nil: increment `summary.Succeeded`.

    6. Set `summary.Elapsed = time.Since(start)`, return summary.

    `downloadWithRetry(ctx, book, backoff)` method:
    1. For attempt 0..MaxRetries:
       a. Call `db.UpdateDownloadStart(p.db, book.ASIN)`.
       b. Create per-ASIN staging subdir: `filepath.Join(cfg.StagingDir, book.ASIN)`.
       c. Call `p.client.Download(ctx, book.ASIN, asinStagingDir)`.
       d. On success: glob staging dir for audio files, call `VerifyM4A` on each, call `MoveToLibrary`, call `db.UpdateDownloadComplete`. Return nil.
       e. On `*audible.AuthError`: return immediately (don't retry auth failures).
       f. On `*audible.RateLimitError`: use `backoff.Delay(attempt) * 2` for longer wait, then retry.
       g. On other error: use `backoff.Delay(attempt)`, then retry.
       h. Between retries, check ctx.Done().
    2. After exhausting retries: call `db.UpdateDownloadError(p.db, book.ASIN, maxRetries, lastErr.Error())`. Return lastErr.

    For progress reporting: call `p.progress.PrintBookProgress(i+1, total, book.Author, book.Title, book.ASIN, -1)` before each download attempt. The -1 means no percentage (audible-cli progress parsing is best-effort, handled in future enhancement).
