```sql
CREATE TABLE IF NOT EXISTS plans (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL,
    description TEXT NOT NULL DEFAULT '',
    status TEXT NOT NULL DEFAULT 'draft',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS plan_operations (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    plan_id INTEGER NOT NULL REFERENCES plans(id),
    seq INTEGER NOT NULL,
    op_type TEXT NOT NULL,
    source_path TEXT NOT NULL,
    dest_path TEXT NOT NULL DEFAULT '',
    status TEXT NOT NULL DEFAULT 'pending',
    error_message TEXT NOT NULL DEFAULT '',
    completed_at DATETIME,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS audit_log (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    entity_type TEXT NOT NULL,
    entity_id TEXT NOT NULL,
    action TEXT NOT NULL,
    before_state TEXT NOT NULL DEFAULT '',
    after_state TEXT NOT NULL DEFAULT '',
    success INTEGER NOT NULL DEFAULT 1,
    error_message TEXT NOT NULL DEFAULT '',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);
```

Create `internal/db/plans.go` in package db:

1. Define validation slices:
   - `ValidPlanStatuses = []string{"draft", "ready", "running", "completed", "failed", "cancelled"}`
   - `ValidOpTypes = []string{"move", "flatten", "split", "delete", "write_metadata"}`
   - `ValidOpStatuses = []string{"pending", "running", "completed", "failed", "skipped"}`

2. Define helper functions:
   - `isValidPlanStatus(status string) bool` -- loop through ValidPlanStatuses
   - `isValidOpType(opType string) bool` -- loop through ValidOpTypes
   - `isValidOpStatus(status string) bool` -- loop through ValidOpStatuses

3. Define `Plan` struct:
   - ID int64
   - Name string
   - Description string
   - Status string
   - CreatedAt time.Time
   - UpdatedAt time.Time

4. Define `PlanOperation` struct:
   - ID int64
   - PlanID int64
   - Seq int
   - OpType string
   - SourcePath string
   - DestPath string
   - Status string
   - ErrorMessage string
   - CompletedAt *time.Time (nullable, use sql.NullTime for scanning)
   - CreatedAt time.Time
   - UpdatedAt time.Time

5. `CreatePlan(db *sql.DB, name, description string) (int64, error)`:
   - INSERT INTO plans (name, description) VALUES (?, ?)
   - Get LastInsertId, return it
   - Also write audit entry: call LogAudit with EntityType "plan", EntityID fmt.Sprintf("%d", id), Action "create", AfterState json.Marshal(map[string]string{"name": name, "description": description, "status": "draft"}), Success true
   - Wrap errors with `fmt.Errorf("create plan: %w", err)`

6. `GetPlan(db *sql.DB, id int64) (*Plan, error)`:
   - SELECT id, name, description, status, created_at, updated_at FROM plans WHERE id = ?
   - Return nil, nil if sql.ErrNoRows
   - Wrap errors with `fmt.Errorf("get plan %d: %w", id, err)`

7. `ListPlans(db *sql.DB, status string) ([]Plan, error)`:
   - If status == "": SELECT all ORDER BY created_at DESC
   - If status != "": SELECT WHERE status = ? ORDER BY created_at DESC
   - Return empty slice (not nil) when no rows
   - Wrap errors with `fmt.Errorf("list plans: %w", err)`

8. `UpdatePlanStatus(db *sql.DB, id int64, status string) error`:
   - Validate status with isValidPlanStatus, return `fmt.Errorf("invalid plan status %q", status)` if invalid
   - UPDATE plans SET status = ?, updated_at = CURRENT_TIMESTAMP WHERE id = ?
   - Check RowsAffected, return error if 0: `fmt.Errorf("plan %d not found", id)`

9. `UpdatePlanStatusAudited(db *sql.DB, id int64, newStatus string) error`:
   - Validate newStatus with isValidPlanStatus
   - GetPlan to capture before state (return error if plan nil)
   - db.Begin() transaction
   - tx.Exec UPDATE plans SET status = ?, updated_at = CURRENT_TIMESTAMP WHERE id = ?
   - Create audit entry: beforeJSON = json.Marshal(map[string]string{"status": plan.Status}), afterJSON = json.Marshal(map[string]string{"status": newStatus})
   - Call LogAuditTx(tx, AuditEntry{EntityType: "plan", EntityID: fmt.Sprintf("%d", id), Action: "status_change", BeforeState: string(beforeJSON), AfterState: string(afterJSON), Success: true})
   - tx.Commit()
   - On any error: tx.Rollback() and return wrapped error

10. `AddOperation(db *sql.DB, op PlanOperation) (int64, error)`:
    - Validate op.OpType with isValidOpType, return `fmt.Errorf("invalid op type %q", op.OpType)` if invalid
    - Verify plan exists: GetPlan(db, op.PlanID), return error if nil (enforcing FK in Go per research pitfall 1)
    - INSERT INTO plan_operations (plan_id, seq, op_type, source_path, dest_path) VALUES (?, ?, ?, ?, ?)
    - Return LastInsertId
    - Wrap errors with `fmt.Errorf("add operation: %w", err)`

11. `ListOperations(db *sql.DB, planID int64) ([]PlanOperation, error)`:
    - SELECT all columns FROM plan_operations WHERE plan_id = ? ORDER BY seq ASC
    - Handle CompletedAt as sql.NullTime
    - Return empty slice (not nil) when no rows
    - Wrap errors with `fmt.Errorf("list operations for plan %d: %w", planID, err)`

12. `UpdateOperationStatus(db *sql.DB, id int64, status, errorMsg string) error`:
    - Validate status with isValidOpStatus, return `fmt.Errorf("invalid op status %q", status)` if invalid
    - If status == "completed": UPDATE SET status = ?, error_message = ?, completed_at = CURRENT_TIMESTAMP, updated_at = CURRENT_TIMESTAMP
    - Else: UPDATE SET status = ?, error_message = ?, updated_at = CURRENT_TIMESTAMP
    - Check RowsAffected, return error if 0
    - Wrap errors with `fmt.Errorf("update operation status %d: %w", id, err)`
